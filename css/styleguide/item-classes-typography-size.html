1.1.1 Text Size

[ .minion | .text-minion | .min | .text-min ] -> 60%
[ .half | .text-half ] -> 75%

[ .normal | .text-normal ] -> 100%
[ .medium | .text-medium ] -> 125%

[ .double | .text-double | .max | .text-max ] -> 200%
[ .triple | .text-triple ] -> 300%
[ .quadruple | .text-quadruple ] -> 400%
